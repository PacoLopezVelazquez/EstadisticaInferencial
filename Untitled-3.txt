#include <stdio.h>

double seno(double x) {
    double sum = 0.0;
    double term = x;
    int sign = 1;
    int n = 1;

    while (1) {
        sum += term;
        term *= -x * x / (n * (2 * n + 1));
        if (sign * term < 1e-10) {
            break;
        }
        sign *= -1;
        n += 2;
    }

    return sum;
}
